**추가:**
```python
from pydantic import BaseModel


class GameSessionSaveRequest(BaseModel):
    score: int
    lives: int
    stage: int


class GameSessionResponse(BaseModel):
    user_id: int
    score: int
    lives: int
    stage: int
    updated_at: str


class GameSessionDeleteResponse(BaseModel):
    message: str
```

**변경 의도:** PUT 요청 바디, GET/PUT 응답, DELETE 응답을 Pydantic 타입으로 정의. `updated_at`은 Redis에 ISO 문자열로 저장하므로 `str` 타입 사용.

### 3. `backend/app/api/v1/game_sessions.py` 생성

**추가:**
```python
import json
from datetime import datetime, timezone
from fastapi import APIRouter, HTTPException
from app.redis_client import redis_client
from app.schemas.game_session import (
    GameSessionSaveRequest,
    GameSessionResponse,
    GameSessionDeleteResponse,
)

REDIS_TTL = 7200  # 2시간
router = APIRouter()


@router.put("/{user_id}", response_model=GameSessionResponse)
def save_game_session(user_id: int, data: GameSessionSaveRequest):
    key = f"game_session:{user_id}"
    payload = {
        **data.model_dump(),
        "user_id": user_id,
        "updated_at": datetime.now(timezone.utc).isoformat(),
    }
    redis_client.set(key, json.dumps(payload), ex=REDIS_TTL)
    return payload


@router.get("/{user_id}", response_model=GameSessionResponse)
def get_game_session(user_id: int):
    key = f"game_session:{user_id}"
    raw = redis_client.get(key)
    if raw is None:
        raise HTTPException(status_code=404, detail="Session not found")
    return json.loads(raw)


@router.delete("/{user_id}", response_model=GameSessionDeleteResponse)
def delete_game_session(user_id: int):
    key = f"game_session:{user_id}"
    redis_client.delete(key)
    return GameSessionDeleteResponse(message="session deleted")
```

**변경 의도:** Redis key `game_session:{user_id}`에 JSON으로 세션 저장·조회·삭제. PUT은 upsert, DELETE는 세션 없어도 200 반환(멱등성), TTL 7200초로 자동 만료.

### 5. `backend/tests/test_game_sessions.py` 생성

**추가:**
```python
import pytest
from fastapi.testclient import TestClient
from app.redis_client import redis_client


TEST_USER_ID = 999


@pytest.fixture(autouse=True)
def cleanup_redis():
    """각 테스트 전후로 테스트용 Redis key 삭제"""
    key = f"game_session:{TEST_USER_ID}"
    redis_client.delete(key)
    yield
    redis_client.delete(key)


def test_put_game_session_saves_data(client):
    response = client.put(
        f"/api/v1/game-sessions/{TEST_USER_ID}",
        json={"score": 1500, "lives": 2, "stage": 3}
    )
    assert response.status_code == 200
    body = response.json()
    assert body["user_id"] == TEST_USER_ID
    assert body["score"] == 1500
    assert body["lives"] == 2
    assert body["stage"] == 3
    assert "updated_at" in body


def test_put_game_session_overwrites_existing(client):
    client.put(f"/api/v1/game-sessions/{TEST_USER_ID}", json={"score": 1000, "lives": 3, "stage": 1})
    response = client.put(f"/api/v1/game-sessions/{TEST_USER_ID}", json={"score": 2500, "lives": 1, "stage": 4})
    assert response.status_code == 200
    body = response.json()
    assert body["score"] == 2500
    assert body["lives"] == 1
    assert body["stage"] == 4


def test_get_game_session_returns_saved_data(client):
    client.put(f"/api/v1/game-sessions/{TEST_USER_ID}", json={"score": 3000, "lives": 2, "stage": 2})
    response = client.get(f"/api/v1/game-sessions/{TEST_USER_ID}")
    assert response.status_code == 200
    body = response.json()
    assert body["user_id"] == TEST_USER_ID
    assert body["score"] == 3000
    assert body["lives"] == 2
    assert body["stage"] == 2


def test_get_game_session_returns_404_when_not_found(client):
    response = client.get(f"/api/v1/game-sessions/{TEST_USER_ID}")
    assert response.status_code == 404


def test_delete_game_session_removes_data(client):
    client.put(f"/api/v1/game-sessions/{TEST_USER_ID}", json={"score": 500, "lives": 3, "stage": 1})
    client.delete(f"/api/v1/game-sessions/{TEST_USER_ID}")
    response = client.get(f"/api/v1/game-sessions/{TEST_USER_ID}")
    assert response.status_code == 404


def test_delete_game_session_returns_success_message(client):
    client.put(f"/api/v1/game-sessions/{TEST_USER_ID}", json={"score": 500, "lives": 3, "stage": 1})
    response = client.delete(f"/api/v1/game-sessions/{TEST_USER_ID}")
    assert response.status_code == 200
    assert response.json() == {"message": "session deleted"}


def test_delete_game_session_is_idempotent(client):
    response = client.delete(f"/api/v1/game-sessions/{TEST_USER_ID}")
    assert response.status_code == 200
    assert response.json() == {"message": "session deleted"}
```

**변경 의도:** TDD RED → GREEN 사이클로 구현. 실제 Redis 컨테이너 연결, `autouse` fixture로 테스트 전후 key 자동 삭제하여 격리.
