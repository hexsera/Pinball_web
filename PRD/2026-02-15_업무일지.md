**변경 후:**
```jsx
import { useState, useEffect } from 'react';
import axios from 'axios';

function HeaderUserInfo() {
  const [role, setRole] = useState(null);

  useEffect(() => {
    if (isLoggedIn && user?.id) {
      axios.get(`/api/v1/users/${user.id}`)
        .then(res => setRole(res.data.role))
        .catch(() => setRole(null));
    }
  }, [isLoggedIn, user?.id]);

  const isAdmin = role === 'admin';
  const pageNavLabel = location.pathname.startsWith('/admin')
    ? '메인 페이지 이동'
    : '관리 페이지 이동';

  // admin에게만 조건부 렌더링
  {isAdmin && (
    <MenuItem onClick={handlePageNavigation}>{pageNavLabel}</MenuItem>
  )}
}
```

**특징:**
- `useEffect`로 컴포넌트 마운트 시 `GET /api/v1/users/{id}` 호출하여 최신 role 획득
- `isAdmin` 변수로 admin 여부 판단, 조건부 렌더링으로 일반 유저에게 항목 숨김
